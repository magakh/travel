async function verifyPhotosAgainstStorage() { showToast('Checking Storage...'); try { const { data: files, error } = await supa.storage.from(PHOTO_BUCKET).list(TRIP_ID + '/', { limit: 1000 }); if (error) { showToast('Cannot list Storage: ' + error.message, true); return; } const existingFiles = new Set((files || []).map(f => f.name)); const appUrlCount = days.reduce((n, d) => n + (d.stops||[]).reduce((m, s) => m + (s.userPhotos||[]).filter(p => p && p.startsWith('http')).length, 0), 0); if (existingFiles.size === 0 && appUrlCount > 0) { showToast('Storage returned empty - need LIST policy in Supabase SQL Editor: CREATE POLICY "trip-photos-list" ON storage.objects FOR SELECT TO anon USING (bucket_id = 'trip-photos');', true); return; } console.log('Storage files:', existingFiles.size, [...existingFiles]); let removed = 0, kept = 0; days.forEach(day => (day.stops||[]).forEach(stop => { if (!stop.userPhotos) return; stop.userPhotos = stop.userPhotos.filter(p => { if (!p || p.startsWith('data:')) return true; const fn = _fingerprintFromUrl(p); if (!fn) return true; if (!existingFiles.has(fn + '.jpg')) { removed++; return false; } kept++; return true; }); })); if (removed > 0) { saveToStorage(); schedulePush(); renderStopList(); renderDetail(); showToast('Removed ' + removed + ' dead link(s), kept ' + kept + ' verified photos'); } else { showToast('All ' + kept + ' photos verified against ' + existingFiles.size + ' Storage files'); } } catch(e) { showToast('Verify failed: ' + e.message, true); } }
